package comments;

public class Statements {
	
	public static void foo() {
		// before a variable declaration
		int a = 42;
		
		// before try
		try {
			// before a method invocation
			// to bar
			bar(a);
		} catch (Exception e) {
		}
		
		// before a for
		for (int i=0; i<10; ++i) {
			// before an if statement
			if (0 == i % 2) {
				// before a continue
				continue;
			}		
			// before a break
			break;
		}
		
		// before while
		while (a > 0) {
			// before expression;
			--a;
		}
	}
	
	public static int bar(int value) {
		if (value == 42) {
			// before throw
			throw new RuntimeException();
		}
		// before a return statement
		return value;
	}
}
