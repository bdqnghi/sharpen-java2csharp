namespace events
{
	public class EventInterfaceImpl : events.EventInterface
	{
		private System.EventHandler<events.QueryEventArgs> _queryStarted;

		public virtual event System.EventHandler<events.QueryEventArgs> QueryStarted
		{
			add
			{
				_queryStarted = (System.EventHandler<events.QueryEventArgs>)System.Delegate.Combine
					(_queryStarted, value);
			}
			remove
			{
				_queryStarted = (System.EventHandler<events.QueryEventArgs>)System.Delegate.Remove
					(_queryStarted, value);
			}
		}
	}
}
