public class Generics {
	static void run() {
		Map<Integer, String> m = new HashMap<Integer, String>();
		for (Map.Entry<Integer, String> e : m.entrySet()) {
			print(e.getKey());
			print(e.getValue());
		}
		Class<?> klass = m.getClass();
		print(klass);
	}

	}
}
