package innerclasses;

public class NestedThis {

	public static class Env {
	
		public Object get(Object key) {
			return null;
		}
		
		public Env add(final Object key, final Object value) {
			return new Env() {
				public Object get(Object key1) {
					if (key == key1) {
						return value;
					}
					return Env.this.get(key);
				}
			};
		}
	}
}
